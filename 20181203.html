def subGroupAnalysis(subgroup):
    global result
    dat = result.loc[result.KMeanclustersTerms == subgroup]
    sg1 = dat.dropna(subset=['Lemtext'])
    sg1 = sg1[['Lemtext', 'KMeanclustersTerms']]
    cv0_sg1 = TfidfVectorizer(binary=False, lowercase = True, stop_words = newstopword_final, use_idf = False, min_df=.01, 
                              max_df =.80, ngram_range = (2,3)) 
    cv0_dm_sg1 = cv0_sg1.fit_transform(sg1['Lemtext']) # document matric
    review_dm_sg1 = cv0_dm_sg1.toarray()
    terms = cv0_sg1.get_feature_names()
    print(subgroup + "Key Words" ,terms, '\n')
    My_k = 8
    km_sg1 = KMeans(n_clusters=My_k, init='k-means++', max_iter=100, random_state = 50)
    review_k_sg1 = km_sg1.fit(review_dm_sg1) # review cluster
#     print (review_k_sg1)
    clusters_sg1 = km_sg1.labels_.tolist()
    #     print (clusters_sg1)
    sg1['KMeanclusters_SubGroup_'+ subgroup] = clusters_sg1
    k_range = range(1,8)
    k_means_set = [KMeans(n_clusters=k,init='k-means++', max_iter=100, random_state = 50).fit(review_dm_sg1) for k in k_range]
    centroids_list_sg1 = [km_result.cluster_centers_ for km_result in k_means_set]
    order_centroids = km_sg1.cluster_centers_.argsort()[:, ::-1]  
    a_list = []
    b_list = []

    for i in range(8):
        a_list.append(i)
        for ind in order_centroids[i, :1]:
            b_list.append(terms[ind])
    sg1_df = pd.DataFrame({'KMean_clusters_sg1': a_list, 'KMean_clustersTerms_sg1': b_list})
    sg1['KMean_clustersTerms_' + subgroup]= sg1['KMeanclusters_SubGroup_'+ subgroup].map(sg1_df.KMean_clustersTerms_sg1)
    df_csv = sg1[['KMeanclusters_SubGroup_' + subgroup,'KMean_clustersTerms_' + subgroup]]
    result = pd.concat([result, df_csv], axis=1)
    result.to_csv('out.csv', sep=',')
